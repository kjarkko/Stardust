256     private void createPlanetButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_createPlanetButtonActionPerformed
257         // TODO add your handling code here:
258         try {
259             double mass = Double.parseDouble(planetMass.getText())
260                     * Math.pow(10, Integer.parseInt(planetMassMagnitude.getText()));
261             double xLoc = Double.parseDouble(planetXLoc.getText())
262                     * Math.pow(10, Integer.parseInt(planetXlocMagn.getText()));
263             double yLoc = Double.parseDouble(planetYloc.getText())
264                     * Math.pow(10, Integer.parseInt(planetYlocMagn.getText()));
265 
266             Planets.get().add(
267                     new Planet(
268                             new Coordinate(xLoc, yLoc),
269                             new Vector(
270                                     Double.parseDouble(planetXvel.getText()),
271                                     Double.parseDouble(planetYvel.getText())),
272                             mass,
273                             Color.decode(planetColor.getText()),
274                             planetName.getText(),
275                             Integer.parseInt(planetRadius.getText())
276                     )
277             );
278         } catch (Exception e) {
279             System.out.println(e.getMessage());
280         }
281     }//GEN-LAST:event_createPlanetButtonActionPerformed

295     private void speedUpActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_speedUpActionPerformed
296         // TODO add your handling code here:
297         int speed = Settings.getSimulationSpeedMultiplier();
298         if (speed <= 100000) {
299             Settings.setSimulationSpeedMultiplier(speed * 10);
300             speedMutiplier = speed * 10 + "";
301             speedMultiplierLabel.setText(speedMutiplier);
302         }
303     }//GEN-LAST:event_speedUpActionPerformed
304 
305     private void speedDownActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_speedDownActionPerformed
306         // TODO add your handling code here:
307         int speed = Settings.getSimulationSpeedMultiplier();
308         if (speed >= 10) {
309             Settings.setSimulationSpeedMultiplier(speed / 10);
310             speedMutiplier = speed / 10 + "";
311             speedMultiplierLabel.setText(speedMutiplier);
312         }
313     }//GEN-LAST:event_speedDownActionPerformed
