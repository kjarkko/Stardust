65      private void drawPlanet(Planet p) {
66          Graphics g = super.getGraphics();
67          g.setColor(p.getColor());
68          
69          int radius = p.radius;
70          Coordinate loc = p.getLocation();
71          int x = 350 - (radius >> 1) + (int)(loc.getX()/Settings.getDistancePerPixel());
72          int y = 350 - (radius >> 1) + (int)(loc.getY()/Settings.getDistancePerPixel());
73          
74          g.fillOval(x, y, radius, radius);
75      }
